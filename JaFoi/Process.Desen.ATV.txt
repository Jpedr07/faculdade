class TaskManager:
    def __init__(self):
        self.tasks = []
    
    def add_task(self, description):
        self.tasks.append({"description": description, "completed": False})
        print("Tarefa adicionada com sucesso!")
    
    def list_tasks(self):
        if not self.tasks:
            print("Nenhuma tarefa cadastrada.")
        else:
            for index, task in enumerate(self.tasks, start=1):
                status = "[X]" if task["completed"] else "[ ]"
                print(f"{index}. {status} {task['description']}")
    
    def complete_task(self, index):
        if 0 < index <= len(self.tasks):
            self.tasks[index - 1]["completed"] = True
            print("Tarefa marcada como concluída!")
        else:
            print("Índice inválido!")
    
    def remove_task(self, index):
        if 0 < index <= len(self.tasks):
            removed_task = self.tasks.pop(index - 1)
            print(f"Tarefa '{removed_task['description']}' removida com sucesso!")
        else:
            print("Índice inválido!")

def main():
    manager = TaskManager()
    while True:
        print("\nGerenciador de Tarefas")
        print("1. Adicionar Tarefa")
        print("2. Listar Tarefas")
        print("3. Concluir Tarefa")
        print("4. Remover Tarefa")
        print("5. Sair")
        opcao = input("Escolha uma opção: ")
        
        if opcao == "1":
            desc = input("Digite a descrição da tarefa: ")
            manager.add_task(desc)
        elif opcao == "2":
            manager.list_tasks()
        elif opcao == "3":
            index = int(input("Digite o número da tarefa a marcar como concluída: "))
            manager.complete_task(index)
        elif opcao == "4":
            index = int(input("Digite o número da tarefa a remover: "))
            manager.remove_task(index)
        elif opcao == "5":
            print("Saindo...")
            break
        else:
            print("Opção inválida! Tente novamente.")

if __name__ == "__main__":
    main()
